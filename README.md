function searchTag ($node, $tag) {
    if ($node->tag == $tag) {                            1
        return $node;
    }
    $arr = $node->children();                            1
    
    foreach ($arr as $child) {                           N
        if (searchTag ($child, $tag) !== null) {         N
            return $child;
        }
    }
}
